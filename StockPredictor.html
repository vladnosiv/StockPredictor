def testStrategy(model, data, get_x):

  class PredictStrategy(Strategy):

      def init(self):
          pass

      def next(self):
          open = self.data.df.Open.tail(1).sum()
          SMA_gaz = self.data.df.Open.tail(WINDOW_SIZE).mean()
          SMA_market = self.data.df.open_market.tail(WINDOW_SIZE).mean()
          SMA_gazneft = self.data.df.open_gazneft.tail(WINDOW_SIZE).mean()

          X = get_x(SMA_gaz, SMA_market, SMA_gazneft)
          predict = model.predict(X)[0]

          if predict > open and not self.position.is_long:
              self.buy(size=.3)
          elif predict < open and not self.position.is_short:
              self.sell(size=.3)

  bt = Backtest(data, PredictStrategy, cash=10000, commission=.002)
  return bt.run()
